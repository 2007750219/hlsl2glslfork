float4x4 world;
float4x4 view;
float4x4 proj;
float4x4 stupid[10];

sampler2D diffuse_map;

struct VS_INPUT {
	float4 position : POSITION;
	float2 uv : TEXCOORD0;
};

struct PS_INPUT {
	float4 position : POSITION;
	float2 uv : TEXCOORD0;
};

PS_INPUT vs_main(VS_INPUT input) {
	PS_INPUT o;

	float3x3 dummy = float3x3(float3(0), float3(0), float3(0));

	float4 wp = mul(input.position, world);
	stupid[2] = view;

	stupid[2]._m30_m31_m32 = dummy._m21_m00_m00;

	o.position = mul(mul(wp, stupid[2]), proj);
	o.uv = input.uv;

	return o;
}

half4 ps_main(PS_INPUT input) : COLOR0 {
	half4 c = half4(1, 1, 1, 1);
	c = tex2D(diffuse_map, input.uv);
	return c;
}
